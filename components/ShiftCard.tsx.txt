}

const DepartmentBadge: React.FC<{ department: Department }> = ({ department }) => {
  const colors: Record<string, string> = {
    [Department.WARD_2A]: 'bg-pink-100 text-pink-800 border-pink-200',
    [Department.WARD_3A]: 'bg-purple-100 text-purple-800 border-purple-200',
    [Department.WARD_3B]: 'bg-blue-100 text-blue-800 border-blue-200',
    [Department.WARD_4A]: 'bg-orange-100 text-orange-800 border-orange-200',
    [Department.OTHER]: 'bg-gray-100 text-gray-800 border-gray-200',
  };

  return (
    <span className={`text-xs px-2 py-1 rounded-full border font-medium ${colors[department] || 'bg-gray-100 text-gray-800'}`}>
      {department}
    </span>
  );
};

const JobRoleBadge: React.FC<{ role: JobRole }> = ({ role }) => {
  const isNurse = role === JobRole.NURSE;
  return (
    <span className={`text-xs px-2 py-1 rounded font-bold ml-2 ${
      isNurse ? 'bg-indigo-600 text-white' : 'bg-teal-600 text-white'
    }`}>
      {role}
    </span>
  );
};
